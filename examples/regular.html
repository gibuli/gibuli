常用正则
匹配数字：/^[0-9]*$/
匹配手机号码：/^1[34578]\d{9}$/
匹配电话号码：/[0-9]{5,}$/
匹配座机：/^\d{3,4}-\d{7,8}(-\d{3,4})?$/
匹配银行账号：/^\d{16,19}$/
匹配邮箱：/\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*/
匹配网址：/(([-a-zA-Z0-9:%._/\+~#=\u4e00-\u9fa5]{1,256}\.[a-zA-Z\u4e00-\u9fa5]{2,}([-a-zA-Z0-9@:%_\+.~#?&//=]*))|(((\d{1,2}|1\d\d|2[0-4]\d|25[0-5])\.){3}(\d{1,2}|1\d\d|2[0-4]\d|25[0-5])))/
匹配QQ：/^[1-9]\d{3,10}$/
匹配文件后缀：/(.jpg|.gif)+(\?|\#|$)/
匹配中文：/^[\u4e00-\u9fa5]{0,}$/
匹配身份证号：/(^\d{15}$)|(^\d{17}([0-9]|X)$)/
匹配IP地址：/^((\d{1,2}|1\d\d|2[0-4]\d|25[0-5])\.){3}(\d{1,2}|1\d\d|2[0-4]\d|25[0-5])$/
匹配首尾空格：/(^\s*)|(\s*$)/g
